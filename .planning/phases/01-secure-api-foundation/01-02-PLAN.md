**Helper functions inside make_app():**

`get_client_ip(request: Request) -> str`:
- Check `request.headers.get("cf-connecting-ip")` first (Cloudflare tunnel)
- Then `request.headers.get("x-forwarded-for")` -- take first IP, strip whitespace
- Then `request.client.host` if request.client exists
- Fallback: return `"unknown"`

`is_rate_limited(ip: str) -> tuple[bool, int]`:
- Get `now = time_mod.time()`
- Get timestamps list from `_rate_limit_store.get(ip, [])`
- Filter to keep only timestamps where `t > now - settings.rate_limit_window_seconds`
- Store filtered list back: `_rate_limit_store[ip] = timestamps`
- If `len(timestamps) >= settings.rate_limit_max`: return `(True, int(timestamps[0] - (now - settings.rate_limit_window_seconds)) + 1)`
- Else: return `(False, 0)`

`record_request(ip: str) -> None`:
- `_rate_limit_store.setdefault(ip, []).append(time_mod.time())`

`_is_localhost(ip: str) -> bool`:
- Return `ip in ("127.0.0.1", "::1", "::ffff:127.0.0.1")`

**The endpoint:**

```python
@app.post("/redesign")
async def redesign_endpoint(body: RedesignRequest, request: Request):
    ip = get_client_ip(request)

    # Rate limit check (localhost exempt)
    if not _is_localhost(ip):
        limited, retry_after = is_rate_limited(ip)
        if limited:
            return JSONResponse(
                status_code=429,
                content={
                    "error": f"Rate limit exceeded. Try again in {retry_after // 60 + 1} minutes.",
                    "retry_after": retry_after,
                },
            )

    # Record attempt (localhost exempt)
    if not _is_localhost(ip):
        record_request(ip)

    # Concurrency check
    if _redesign_semaphore.locked():
        return JSONResponse(
            status_code=429,
            content={"error": "A redesign is already in progress. Please wait and try again."},
        )

    return EventSourceResponse(redesign_stream(body.url))
```

**The SSE generator:**

```python
async def redesign_stream(url: str) -> AsyncGenerator[ServerSentEvent, None]:
    job_id = str(uuid4())
    async with _redesign_semaphore:
        try:
            # Step 1: Crawl
            yield ServerSentEvent(
                data=json.dumps({"job_id": job_id, "message": "Crawling your site...", "progress": 10}),
                event="crawling",
            )
            crawl_result = await crawl(url)
            if crawl_result.error:
                yield ServerSentEvent(
                    data=json.dumps({"job_id": job_id, "error": "Could not reach that website. Check the URL and try again."}),
                    event="error",
                )
                return

            # Step 2: Redesign (sync -- use to_thread)
            yield ServerSentEvent(
                data=json.dumps({"job_id": job_id, "message": "Claude is redesigning...", "progress": 40}),
                event="redesigning",
            )
            html = await asyncio.to_thread(
                redesign, crawl_result, settings.claude_code_api_url, settings.claude_model
            )

            # Sanitize with nh3
            html = nh3.clean(html)

            # Step 3: Deploy (sync -- use to_thread)
            yield ServerSentEvent(
                data=json.dumps({"job_id": job_id, "message": "Deploying your redesign...", "progress": 80}),
                event="deploying",
            )
            result = await asyncio.to_thread(deploy, url, html, settings.sites_dir)

            # Step 4: Done
            yield ServerSentEvent(
                data=json.dumps({
                    "job_id": job_id,
                    "message": "Done!",
                    "progress": 100,
                    "url": f"/{result.subdomain}/",
                    "subdomain": result.subdomain,
                }),
                event="done",
            )
        except Exception:
            yield ServerSentEvent(
                data=json.dumps({"job_id": job_id, "error": "Redesign service unavailable. Please try again later."}),
                event="error",
            )
            return
```

**CRITICAL implementation notes:**
- `crawl()` is async -- `await` directly, do NOT wrap in `asyncio.to_thread()`
- `redesign()` and `deploy()` are sync -- MUST use `asyncio.to_thread()`
- The semaphore is acquired INSIDE the generator (not in the endpoint), so it's held for the duration of the pipeline
- After yielding an `error` event, `return` immediately to close the stream
- The broad `except Exception` catches any unexpected failure (API gateway down, deployer error, etc.) and emits a sanitized error message -- no internal exception details exposed
- nh3.clean() is called on the HTML BEFORE passing to deploy()
- Progress values: crawling=10, redesigning=40, deploying=80, done=100 (per CONTEXT.md)
- The `done` event includes both `url` (relative path like `/acme-corp/`) and `subdomain` (slug like `acme-corp`)

```python
import threading

def test_concurrency_cap(tmp_sites, mock_crawl_result, mock_deploy_result):
    block = threading.Event()
    proceed = threading.Event()

    async def slow_crawl(url):
        proceed.set()  # Signal that first request started
        await asyncio.sleep(0)  # Yield to event loop
        # Wait synchronously (this simulates a long crawl)
        block.wait(timeout=5)
        return mock_crawl_result

    mock_html = "<html><body>OK</body></html>"
    with patch("sastaspace.server.crawl", side_effect=slow_crawl), \
         patch("sastaspace.server.redesign", return_value=mock_html), \
         patch("sastaspace.server.deploy", return_value=mock_deploy_result):
        from sastaspace.server import make_app
        app = make_app(tmp_sites)
        client = TestClient(app)

        results = {}

        def first_request():
            results["first"] = client.post("/redesign", json={"url": "https://example.com"})

        t = threading.Thread(target=first_request)
        t.start()
        proceed.wait(timeout=5)  # Wait for first request to start

        # Second request should get 429
        resp = client.post("/redesign", json={"url": "https://example.com"})
        assert resp.status_code == 429
        assert "already in progress" in resp.json()["error"]

        block.set()  # Let first request finish
        t.join(timeout=5)
```

**SSE parsing helper** -- add at the top of the test file:
```python
def parse_sse_events(text: str) -> list[dict]:
    """Parse SSE text into list of {event, data} dicts."""
    events = []
    current = {}
    for line in text.split("\n"):
        if line.startswith("event:"):
            current["event"] = line.split(":", 1)[1].strip()
        elif line.startswith("data:"):
            raw = line.split(":", 1)[1].strip()
            try:
                current["data"] = json.loads(raw)
            except json.JSONDecodeError:
                current["data"] = raw
        elif line == "" and current:
            events.append(current)
            current = {}
    if current:
        events.append(current)
    return events
```
