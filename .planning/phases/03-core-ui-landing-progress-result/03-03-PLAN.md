```typescript
import type { Metadata } from "next";
import { ResultView } from "@/components/result/result-view";

// Dynamic metadata (per RESULT-04)
export async function generateMetadata({
  params,
}: {
  params: Promise<{ subdomain: string }>;
}): Promise<Metadata> {
  const { subdomain } = await params;
  const domain = subdomain.replace(/-/g, ".");
  return {
    title: "Your redesign is ready -- SastaSpace",
    description: `${domain} has been redesigned by AI`,
  };
}

export default async function ResultPage({
  params,
}: {
  params: Promise<{ subdomain: string }>;
}) {
  const { subdomain } = await params;
  return <ResultView subdomain={subdomain} isShareable />;
}
```

Key implementation details:
- `params` is typed as `Promise<{ subdomain: string }>` and MUST be awaited (Next.js 16 breaking change, per research pitfall 2)
- The page is a server component -- it passes `subdomain` and `isShareable={true}` to the client-side `ResultView`
- `isShareable` is `true` because this route is accessed directly (not navigated from progress flow)
- Title per UI-SPEC copywriting contract: `"Your redesign is ready -- SastaSpace"` (uses `--` em dash equivalent)
- Description per UI-SPEC: `"{domain} has been redesigned by AI"`
- Domain reconstruction from subdomain slug: `subdomain.replace(/-/g, ".")`
- Do NOT add `"use client"` -- this must be a server component so generateMetadata works
