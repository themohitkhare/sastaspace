```typescript
export function validateUrl(input: string): { valid: boolean; url: string; error?: string }
```

- Use `new URL(url)` constructor to parse
- Reject if hostname has no dot (no TLD)
- On success: `{ valid: true, url: parsed.href }`
- On failure: `{ valid: false, url: input, error: "Please enter a valid website address" }`

```typescript
export function extractDomain(url: string): string
```
- Parse URL (prepending `https://` if needed)
- Return `parsed.hostname` with `www.` prefix removed
- On error, return the input as-is
